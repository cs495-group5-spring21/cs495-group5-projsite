import { Injectable, NgZone } from '@angular/core';
import { Router } from '@angular/router';
import { FirebaseUser } from 'src/app/interfaces/firebase-user';
import { AngularFireAuth } from '@angular/fire/auth';
import { AngularFirestore, AngularFirestoreDocument } from '@angular/fire/firestore';
import 'firebase/auth';
import 'firebase/firestore';
import { Subscription } from 'rxjs';
import { AnalyticsService } from './analytics.service';


@Injectable({
  providedIn: 'root'
})

export class FirebaseAuthService {

  userData: any; //data stored by firebase authentication
  userDoc: FirebaseUser = {
    uid: "null",
    email: "null",
    emailVerified: false,
  };

  authStateSubscription!: Subscription;

  constructor(
    public afs: AngularFirestore,
    public afAuth: AngularFireAuth,
    public router: Router,
    public ngZone: NgZone,
    private analytics: AnalyticsService
  ) {

    this.GetAuthUserState().then((user) => {
      this.UpdateLocalStorageUser(user);
    })
    //console.log("constructor called on db auth service");
  }

  /**
   * Function called when user signs in using email and password
   * 
   * @param email the email that the user used to register their account
   * @param password the password linked to the email
   */
  SignIn(email: string, password: string) {
    return this.afAuth.signInWithEmailAndPassword(email, password)
      .then((result) => {

        this.UpdateLocalStorageUser(result.user);

        this.UpdateUserDocData({ 'lastLogin': Date.now() }); //update for lastLogin
        // console.log("returning from firebase auth signin", result);

        this.analytics.track({
          'event': "userSignIn",
          'timestamp': Date.now(),
          'userId': this.userData.uid
        });

        this.router.navigate(['archive']);
      });
  }

  /**
   * Function called when user signs up using email and password
   * 
   * @param email the email that the user is using to register their account
   * @param password the password linked to the email
   */
  SignUp(email: string, password: string) {
    return this.afAuth.createUserWithEmailAndPassword(email, password)
      .then((result) => {

        //console.log("returning from firebase auth create user: ", result);
        this.UpdateLocalStorageUser(result.user);

        //console.log("is logged in: ", this.isLoggedIn);
        // console.log("user data post sign up: ", this.userData);

        //create user document and send verification email
        this.PutUserDocData(this.userData);
        this.SendVerificationMail();

        this.analytics.track({
          'event': 'userSingUp',
          'timestamp': Date.now(),
          'userId': this.userData.uid
        });

        this.router.navigate(['auth/verify-email']);

      });
  }

  /**
   * Function called to send a verification email to the user when 
   * they initially sign up or they request a new verification email
   */
  SendVerificationMail() {
    return this.GetAuthUserState().then(currentUser => currentUser!.sendEmailVerification());
  }

  /**
   * Function called when user clicks the forgot password button
   * 
   * @param passwordResetEmail email to send the forgot password request to
   */
  ForgotPassword(passwordResetEmail: string) {
    return this.afAuth.sendPasswordResetEmail(passwordResetEmail);
  }

  /**
   * Function called to check if user is currently logged in
   */
  get isLoggedIn(): boolean {
    var userJson = localStorage.getItem('user');
    const user = userJson !== null ? JSON.parse(userJson) : null;

    //console.log("returning isLoggedIn as " + ((user !== null) ? true : false));

    return (user !== null) ? true : false;
  }


  /**
   * Function called to check if current user's emial is verified
   */
  get isVerified(): boolean {
    var userJson = localStorage.getItem('user');
    const user = userJson !== null ? JSON.parse(userJson) : null;

    return user !== null ? user.emailVerified : false;
  }

  /**
   * Function called to get the id of the current user
   */
  get id(): string {
    var userJson = localStorage.getItem('user');
    const user = userJson !== null ? JSON.parse(userJson) : null;
    return user.uid; //TODO: This can cause a bug if the user happens to be null. i don't think this ever happens tho. 
  }

  /**
   * Function called when user signs up to create a document in firestore with their initial data
   * 
   * @param user Object containing the information of the current user
   */
  PutUserDocData(user: any) {
    //console.log("SetUserData Called")
    const userRef: AngularFirestoreDocument<any> = this.afs.doc('users/' + user.uid);
    const userData: FirebaseUser = {
      uid: user.uid,
      email: user.email,
      displayName: user.displayName,
      emailVerified: user.emailVerified,
      initialized: false,
      admin: false,
      lastLogin: Date.now()
    }
    return userRef.set(userData, {
      merge: true
    })
  }

  /**
   * Function called to update data in the user's Firestore document
   * 
   * @param data Object containing whatever data is to be updated in the user's Firestore document.
   */
  UpdateUserDocData(data: any) {
    //this.userData.reload();
    const userRef: AngularFirestoreDocument<any> = this.afs.doc('users/' + this.userData.uid);
    //console.log("trying to update user: " + this.userData.uid + " with the following data: ", data);

    userRef.update(data).catch((error) => {
      console.log('In update user data: ', error);
      console.log("above error called trying to update user: " + this.userData.uid + " with the following data: ", data);
      //console.log('User Data after being updated: \n', JSON.stringify(this.userData));
    });
  }

  /**
   * Function called when user logs out
   */
  SignOut() {
    return this.afAuth.signOut().then(() => {
      this.userData = null;
      localStorage.removeItem('user');
    })
  }

  /**
   * Function called to get the current state of Firebase Authentication
   */
  GetAuthUserState() {
    return this.afAuth.currentUser;
  }

  /**
   * Function called to update the user data stored in local storage
   * 
   * @param user data of the current user
   */
  UpdateLocalStorageUser(user: any) {

    if (user == undefined || user == null) {
      return;
    }

    //console.log("updating local storage user with: ", user);
    this.userData = user;
    localStorage.setItem('user', JSON.stringify(this.userData));
  }

}
