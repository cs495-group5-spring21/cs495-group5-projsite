import { Injectable } from '@angular/core';
import { AngularFirestore, DocumentReference } from '@angular/fire/firestore';
import { Observable } from 'rxjs';
import { Measure } from '../interfaces/measure';
import { AnalyticsService } from './analytics.service';

@Injectable({
  providedIn: 'root'
})
export class MeasuresService {

  constructor(private db: AngularFirestore, private analyticsService: AnalyticsService) { }

  private measures!: Measure[];

  /**
   * Function called to get all Measures from the measures collection in Firestore
   * 
   * @returns an Observable containing and array of Measures
   */
  getMeasures(): Observable<Measure[]> {
    return this.db.collection<Measure>('measures').valueChanges({ idField: 'id' });
  }

  //TODO: LINK ID VS AUTO GENERATED ID DEBATE
  /**
   * Function called to get a single measure.
   * 
   * @param id Measure document ID to get
   * @returns an Observable containing a single Measure
   */
  getMeasure(id: string | undefined): Observable<Measure | undefined> {
    return this.db.collection<Measure>('measures').doc(id).valueChanges();
  }

  /**
   * Function called to create a Firestore document containing the data of a measure.
   * 
   * @param data Object containing whatever data is to be updated in the user's Firestore document.
   */
  createMeasure(data: any): Promise<DocumentReference<Measure>> {
    return this.db.collection<Measure>('measures').add(data);
  }

  /**
   * Function called to delete a single measure.
   * 
   * @param id Measure document ID to be deleted
   * @returns a void Promise
   */
  deleteMeasure(id: any): Promise<void>  {
    return this.db.collection<Measure>('measures').doc(id).delete().then(() => {
      this.analyticsService.deleteMeasureAnalytics(id);
    });
  }

  /**
   * Updates a measure in the database
   * 
   * @param id Measure document ID to update
   * @param data Data to put at the location of this measure.
   * @returns Update promise that can be subscribed to.
   */
  updateMeasure(id: any, data: any): Promise<void> {
    const docRef = this.db.collection<Measure>('measures').doc(id);
    return docRef.update(data);
  }

  //Should limit to 40 characters each for overflow reasons
  /**
   * Function to get the current list of tags fro the measures database
   * 
   * @returns list of tags
   */
  getTags() {
    return [
      'Activities', 
      'Advanced Care Planning & Decision Making', 
      'Affect', 
      'Caregiving', 
      'Cognition', 
      'Death', 
      'Dementia-Related Behaviors', 
      'Environment', 
      'Ethnicity / Race',
      'Functional Status',
      'Home Safety',
      'Interpersonal Processes',
      'Intrapersonal Processes',
      'Long Term Care',
      'Motivation',
      'Personality',
      'Physical Health',
      'Psychological Disorders',
      'Quality of Life',
      'Religion / Spirituality',
      'Service Utilization',
      'Special Populations',
      'Stress and Coping',
      'Symptoms and Syndromes',
      'Time'
    ];
  }
}
