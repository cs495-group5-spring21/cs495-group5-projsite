import { Injectable } from '@angular/core';
import { AngularFirestore, AngularFirestoreDocument } from '@angular/fire/firestore';
import { Observable } from 'rxjs';
import { FirebaseUser } from 'src/app/interfaces/firebase-user';

@Injectable({
  providedIn: 'root'
})
export class UserService {

  constructor(private db: AngularFirestore, public afs: AngularFirestore) { }

  /**
   * Function called to get a specific document
   * 
   * @param docId name of the document to get
   */
  getUserData(docId: string): Observable<FirebaseUser | undefined>{
    return this.db.collection<FirebaseUser>('users').doc(docId).valueChanges();
  }

  /**
   * Function called to get documents for all of the users
   */
  getUsers(): Observable<FirebaseUser[]>{
    return this.db.collection<FirebaseUser>('users').valueChanges();
  }

  /**
   * Function for adding data to a given user's document
   * 
   * @param userId Id of user whose data is to be updated
   */
  updateSpecificUserData(userId: string, data: any) {
    const userRef: AngularFirestoreDocument<any> = this.afs.doc('users/' + userId);
    //console.log("trying to update user: " + userId + " with the following data: ", data);

    userRef.update(data).catch((error) => {
      console.log('In update user data: ', error);
      console.log("above error called trying to update user: " + userId + " with the following data: ", data);
      //console.log('User Data after being updated: \n', JSON.stringify(this.userData));
    });
  }

  /**
   * Function for deleting a user's document
   * 
   * @param userId Id of user whose data is to be deleted
   */
  deleteUserDoc(userId: string) {
    const userRef: AngularFirestoreDocument<any> = this.afs.doc('users/' + userId);
    userRef.delete().catch((error) => {
      console.log('In delete user data: ', error);
    });
  }
}
