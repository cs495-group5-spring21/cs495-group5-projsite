import { Injectable } from '@angular/core';
import { AngularFirestore } from '@angular/fire/firestore';
import { Observable } from 'rxjs';

import { map } from 'rxjs/operators';
import { Analytic } from '../interfaces/analytic';

@Injectable({
  providedIn: 'root'
})
export class AnalyticsService {

  constructor(private db: AngularFirestore) { }

  /**
   * 
   * Uploads a new document to the analytics collection in the DB. 
   * 
   * @param event A string that is used to identify and group similar events. e.g. 'measureViewed'
   * @param data  Optional data corresponding to the event that was tracked. 
   */
  track(data?: {}) {

    this.db.collection('analytics').add(
      data
    ).then((docRef) => {
      //console.log("successfully uploaded analytics event: ", docRef);
    }).catch((error) => {
      console.error("error uploading analytics event: ", error);
    });
  }

  /**
   * Function called to get analytics of a specific type
   * 
   * @param type type of analytic being created
   * @returns an observable containing anayltics of the given type
   */
  getAnalytics<T>(type: string): Observable<T[]> {
    return this.db.collection<T>('analytics', ref => ref.where('event', '==', type)).snapshotChanges().pipe(
      map(actions => actions.map(a => {
        const data = a.payload.doc.data() as T;
        const id = a.payload.doc.id;
        return { id, ...data };
      }))
    );
  }

  /**
   * This function is run when a measure is deleted. It deletes any analytics linked to that measure.
   * 
   * @param measureId id of the measure that has been deleted
   */
  async deleteMeasureAnalytics(measureId: string) {
    //console.log("Delete Analytics Function")
    // this.getAnalytics<Analytic>('measureViewed')
    var snapshot = await this.db.collection<Analytic>('analytics', ref => ref.where('measureId', '==', measureId)).get();
    snapshot.forEach((doc) => {
      doc.docs.forEach((d) => {
        return this.db.collection<Analytic>('analytics').doc(d.id).delete().catch((error) => {
          //console.log("On Measure Analytic Delete: " + error)
        });
      })
    }).catch((error) => { console.log(error); });
  }
}

export var AnalyticTypes = {
  SIGNIN: "userSignIn"
}
