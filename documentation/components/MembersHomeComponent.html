import { Component, HostListener, OnInit, ViewChild, ElementRef } from '@angular/core';
import { Story } from '../../interfaces/story';
import { DateFormatter } from '../../models/DateFormatter';
import { StoriesService } from 'src/app/services/stories.service';


@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.scss']
})
export class MembersHomeComponent implements OnInit {
  @ViewChild('welcome') welcomeElement!: ElementRef;
  @ViewChild('news') newsElement!: ElementRef;
  @ViewChild('archive') archiveElement!: ElementRef;
  @ViewChild('collaborate') collaborateElement!: ElementRef;
  @ViewChild('nav') navElement!: ElementRef;
  @ViewChild('body') bodyElement!: ElementRef;


  stories: Story[] = [];

  currentActive: number = 1;
  welcomeYOffset: number = 0;
  newsYOffset: number = 0;
  archiveYOffset: number = 0;
  collaborateYOffset: number = 0;

  offsetBottom: number = 0;
  fixLocation: number = 0;

  sidebarSticky: boolean = false;
  sidebarFixedBottom: boolean = false;


  /*https://stackoverflow.com/questions/54446171/angular-6-how-to-highlight-element-on-navbar-depending-on-scroll*/
  constructor(private storiesService: StoriesService) { }

  ngOnInit(): void {
    this.stories = this.storiesService.getStoriesHomePageMembers();
    //console.log(this.stories);
  }

  /**
   * establishes x,y locations of all elements for UI sidebar scroll placement
   */
  ngAfterViewInit() {
    this.welcomeYOffset = this.welcomeElement.nativeElement.offsetTop;
    this.newsYOffset = this.newsElement.nativeElement.offsetTop;
    this.archiveYOffset = this.archiveElement.nativeElement.offsetTop;
    this.collaborateYOffset = this.collaborateElement.nativeElement.offsetTop;

    this.offsetBottom = this.collaborateYOffset + 1.5 * this.collaborateElement.nativeElement.offsetHeight;
    this.fixLocation = this.bodyElement.nativeElement.offsetHeight - (this.navElement.nativeElement.offsetHeight / 2);

    //console.log("body height: " + this.bodyElement.nativeElement.offsetHeight);
    //console.log("sidebar offset /2: " + this.navElement.nativeElement.offsetHeight / 2);

  }

  getDate(story: Story): string {
    return DateFormatter.getDate(story);
  }

  getBottomFix(): string {
    //console.log("I RAN!");
    return this.fixLocation + 'px';
  }

  /**
   * scrolls to element passed
   * @param el the element to be scrolled to
   */
  scroll(el: HTMLElement) {
    el.scrollIntoView({ behavior: "smooth", block: "end" });
  }

  /**
   * scrolls to bottom of page for contact
   */
  scrollToBottom() {
    window.scrollTo({ top: document.body.scrollHeight, behavior: "smooth" });
  }


  @HostListener('window:scroll', [])

  /**
   * determines how to display the sidebar
   */
  scrolling() {
    //console.log("scrolling offset: " + window.pageYOffset);

    if (window.pageYOffset > this.collaborateYOffset) this.currentActive = 4;
    else if (window.pageYOffset > this.archiveYOffset) this.currentActive = 3;
    else if (window.pageYOffset > this.newsYOffset) this.currentActive = 2;
    else this.currentActive = 1;

    if (window.pageYOffset >= 420) this.sidebarSticky = true;
    else this.sidebarSticky = false;

    if (window.pageYOffset >= this.offsetBottom) {
      this.sidebarFixedBottom = true;
      this.sidebarSticky = false;
      this.fixLocation = this.bodyElement.nativeElement.offsetHeight - (this.navElement.nativeElement.offsetHeight) - 60;
    }
    else this.sidebarFixedBottom = false;
    //console.log("on " + this.currentActive);
  }

}
