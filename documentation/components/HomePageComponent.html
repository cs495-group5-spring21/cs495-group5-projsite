import { Component, OnInit } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { SharedModule } from '../../shared/shared.module';
import { Story } from '../../interfaces/story';
import { DateFormatter } from '../../models/DateFormatter';
import { StoriesService } from '../../services/stories.service'
import { Observable, Subscription } from 'rxjs';

@Component({
  selector: 'app-home-page',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.scss']
})
export class HomePageComponent implements OnInit {
  stories:Story[] = [];
  storiesSub?: Subscription;

  constructor(private titleService : Title, private storiesService : StoriesService) { 
  }

  /**
   * On initialization of the page, sets the title of the home page and 
   * loads the empty array 'stories' with every story who's postPublic variable is set to true inside the database.
   */
  ngOnInit(): void {
    this.titleService.setTitle("HOME | Alabama Research Institute on Aging");
    this.stories = this.storiesService.getStoriesHomePage();
       
  }

  /**
   * 
   */
  ngOnDestory() {
    this.storiesSub?.unsubscribe();
  }
  
  /**
   * Retrieves a story from the database and returns a formatted date of the story retrieved.
   * @param story A document stored in the database under the stories collection.
   * @returns Returns the formatted date of the document in a string to be displayed on the home page.
   */
  getDate(story: Story) : string {
    return DateFormatter.getDate(story);
  }

}
