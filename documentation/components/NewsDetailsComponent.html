import { Component, OnInit } from '@angular/core';
import { StoriesService } from 'src/app/services/stories.service';
import { ActivatedRoute, Router } from '@angular/router';
import { Story } from 'src/app/interfaces/story'
import { Observable, Subscription } from 'rxjs';
import { AngularFireStorage } from '@angular/fire/storage';
import { take } from 'rxjs/operators';
import { DateFormatter } from 'src/app/models/DateFormatter';

@Component({
  selector: 'app-news-details',
  templateUrl: './news-details.component.html',
  styleUrls: ['./news-details.component.scss']
})
export class NewsDetailsComponent implements OnInit {

  public story!: Observable<Story | undefined>;
  public storyID: string = "";
  private storySub!: Subscription;
  private urlSub!: Subscription;
  
  public tags: String = "";
  public date!: Date;
  public imgURL: String = "";

  constructor( private storiesService: StoriesService, 
    private _activatedRoute: ActivatedRoute,
    private router: Router,
    private storage: AngularFireStorage
    ) { }

  ngOnInit(): void {
    this._activatedRoute.paramMap.subscribe(params => {
        this.storyID = params.get('storyID')!;
        //console.log(this.storyID);

        this.story = this.storiesService.getStory(this.storyID);
        //console.log(this.story);
        //if(this.story) this.router.navigate(['/news']);
    });

    this.storySub = this.story.subscribe((story) =>{
        if(story) {
          this.tags = this.getTags(story.tags);

          var d8: any = story.date;
          this.date = d8.toDate();

          var u = story.img;
          story.img = "";

          if(u) this.urlSub = this.storage.refFromURL('gs://ariawebsitedb.appspot.com/' + u).getDownloadURL().pipe(take(1)).subscribe((url) =>{
            story.img = url;
            this.imgURL = url;
          });
        }
        else this.router.navigate(['/news']);
        
    })
  }

  /**
   * parses tags from array to string
   * @param tags the array to be parsed
   * @returns parsed string of tags
   */
  getTags(tags: string[]) : string {
    let output: string = "";
    output += tags[0];
    if(tags.length > 1) output += ", " + tags[1];
    return output;
  }

  ngOnDestroy() {
    this.storySub.unsubscribe();
    if(this.urlSub) this.urlSub.unsubscribe();
  }

  /**
   * uses DateFormatter to parse date if exists, if not fetched yet, parses nothing
   * @returns string of parsed date
   */
  getDate(thedate:Date): String{
    if (thedate) {
      var temp: Story = {
        id: "",
        date: thedate,
        title: "",
        tags: [],
        slug: "",
        body: "",
        img: "",
        postPublic: true,
        postMembers: true
      };

      let output = DateFormatter.getDate(temp);
      //console.log(output);
      return output;
    }

    else return "";
  }
}
