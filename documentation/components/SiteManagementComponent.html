import { Component, OnInit, ViewChild, ChangeDetectorRef, ElementRef } from '@angular/core';
import { MatPaginator } from '@angular/material/paginator';
import { MatTableDataSource } from '@angular/material/table';
import { Story } from "src/app/interfaces/story";
import { FormGroup, FormControl, FormBuilder, Validators } from '@angular/forms';
import { AngularFireStorage } from '@angular/fire/storage';
import { Observable, Subscription } from 'rxjs';
import { StoriesService } from 'src/app/services/stories.service';
import { formatDate } from '@angular/common';

@Component({
  selector: 'app-sitemanagement',
  templateUrl: './site-management.component.html',
  styleUrls: ['./site-management.component.scss']
})
export class SiteManagementComponent implements OnInit {

  createPostForm !: FormGroup;
  submitted: boolean = false;
  prevImgURL: any;
  file: any = null;
  public uploadProgress!: Observable<number | undefined>;
  public uploading: boolean = false;
  public editing: boolean = false;
  public startDate = new Date();

  public editedStoryOrig: any = null;

  storiesSub!: Subscription;

  currentArticles: MatTableDataSource<Story> = new MatTableDataSource<Story>();;
  displayedColumns: string[] = ['date', 'title', 'slug', 'postedOn', 'edit'];

  editorOptions = {
    toolbar: {
      container: [
        ["bold", "italic", "underline"],
        [{ 'list': 'ordered' }, { 'list': 'bullet' }],
        [{ 'indent': '-1' }, { 'indent': '+1' }],
        [{ 'header': [3, 4, 5, 6, false] }],
        ['link'],
        [{ 'color': [] }],
        [{ 'align': [] }],
        ['clean']
      ]
    }
  };
  placeholderText: string = "Write something here...";

  @ViewChild(MatPaginator) paginator!: MatPaginator;
  @ViewChild('createModal') createModal!: ElementRef;

  constructor(private storiesService: StoriesService, private fb: FormBuilder, private cdr: ChangeDetectorRef, private storage: AngularFireStorage) {
  }

  /**
   * Retrieves the stories from firebase and populates the table of new stories. Also resets form 
   * in modal
   * 
   */
  ngOnInit(): void {

    this.createPostForm = this.fb.group({
      'postTitle': ['', Validators.required],
      'postDate': [formatDate(this.startDate, 'yyyy-MM-dd', 'en')],
      'postSlug': [''],
      'postTags': [''],
      'postImg': [null, Validators.required],
      'editor': new FormControl(null),
      'postOnPublic': [true],
      'postOnMembers': [false]
    });

    this.storiesSub = this.storiesService.getStories().subscribe(things => {
      this.currentArticles.data = things;
      this.currentArticles.paginator = this.paginator;
    });

  }

  /**
   * Unsubscribes from the stories subscription in firebase
   */
  ngOnDestroy() {
    this.storiesSub.unsubscribe();
  }

  /**
   * determines whether a story is being created or updated, and calls appropriate function
   */
  onSubmit() {
    if (this.editing) this.onEditSubmit();
    else this.onCreatePostSubmit();
  }

  /**
   * Notifies modal that user wants to create a new story and resets form if it was in the process of editing one.
   */
  onAdd() {
    if (this.editing) {
      this.resetForm();
    }
  }

  /**
   * creates a new post in firebase from form, uploads the data and photo, and clears the form. While uploading,
   * modal is disabled and loading bar appears.
   */
  onCreatePostSubmit() {

    const path = `storyimg/${Date.now()}_${this.file.name}`;
    const ref = this.storage.ref(path);

    this.uploading = true;
    var task = this.storage.upload(path, this.file);
    this.uploadProgress = task.percentageChanges();
    task.then((i) => {
      this.storiesService.createStory({
        date: new Date(this.createPostForm.get('postDate')?.value),
        title: this.createPostForm.get('postTitle')?.value,
        tags: this.parseTags(this.createPostForm.get('postTags')?.value),
        slug: this.createPostForm.get('postSlug')?.value,
        body: this.createPostForm.get('editor')?.value,
        img: path,
        postPublic: this.createPostForm.get('postOnPublic')?.value,
        postMembers: this.createPostForm.get('postOnMembers')?.value,
      }).then(() => {
        //enable form and close modal
        this.createModal.nativeElement.click();
        this.resetForm();
      });
    });
  }

  /**
   * resets the form to blank
   */
  resetForm() {
    //reset form
    this.uploading = false;
    this.editing = false;
    this.createPostForm = this.fb.group({
      'postTitle': ['', Validators.required],
      'postDate': [formatDate(this.startDate, 'yyyy-MM-dd', 'en')],
      'postSlug': [''],
      'postTags': [''],
      'postImg': [null, Validators.required],
      'editor': new FormControl(null),
      'postOnPublic': [true],
      'postOnMembers': [false]
    });
    this.prevImgURL = null;
    this.file = null;
    this.editedStoryOrig = null;
  }

  /**
   * checks if a new photo was uploaded and uploads new file and deletes old one accordingly. then updates all remaining data in firebase document.
   */
  onEditSubmit() {
    //(this.createPostForm);

    if (this.createPostForm.pristine) {
      this.createModal.nativeElement.click(); //close modal, nothing happened
    }
    else { //upload files and update info
      if (this.file) { //upload new photo
        const path = `storyimg/${Date.now()}_${this.file.name}`;
        const ref = this.storage.ref(path);
        this.uploading = true;

        var task = this.storage.upload(path, this.file);
        this.uploadProgress = task.percentageChanges();
        task.then((i) => {
          //delete original photo
          let delRef = this.storage.ref(this.editedStoryOrig.img);
          let deleteTask = delRef.delete().toPromise().then(() => {
            //original photo deleted, update entry
            //console.log(this.editedStoryOrig.img + ' deleted!');

            this.storiesService.updateStory(this.editedStoryOrig.id, {
              date: new Date(this.createPostForm.get('postDate')?.value),
              title: this.createPostForm.get('postTitle')?.value,
              tags: this.parseTags(this.createPostForm.get('postTags')?.value),
              slug: this.createPostForm.get('postSlug')?.value,
              body: this.createPostForm.get('editor')?.value,
              img: path,
              postPublic: this.createPostForm.get('postOnPublic')?.value,
              postMembers: this.createPostForm.get('postOnMembers')?.value,
            }).then(() => {
              //("story updated");
              this.resetForm();
              this.createModal.nativeElement.click();
            });

          })
            .catch((error: any) => {
              //file wasnt deleted
              console.log("file not deleted with error ", error);
            });
        });
      }
      else { //just update data
        this.storiesService.updateStory(this.editedStoryOrig.id, {
          date: new Date(this.createPostForm.get('postDate')?.value),
          title: this.createPostForm.get('postTitle')?.value,
          tags: this.parseTags(this.createPostForm.get('postTags')?.value),
          slug: this.createPostForm.get('postSlug')?.value,
          body: this.createPostForm.get('editor')?.value,
          postPublic: this.createPostForm.get('postOnPublic')?.value,
          postMembers: this.createPostForm.get('postOnMembers')?.value,
        }).then(() => {
          //console.log("story updated");
          this.resetForm();
          this.createModal.nativeElement.click();
        });
      }
    }
  }

  /**
   * image uploading handler for image previews
   */
  onChangePhotoUpload(event: any) {
    var files = event.target.files
    if (files.length == 0) return;
    var reader = new FileReader();
    var prevImgPath = files;
    reader.readAsDataURL(files[0]);
    reader.onload = (_event) => {
      this.prevImgURL = reader.result;
      this.file = files[0];
    }
  }

  /**
   * just a tag parser to format the array of tags into a string
   */
  parseTags(tags: String) {
    let r = RegExp(" *, *");
    let a = tags.split(r);
    return a;
  }

  /**
   * formats the date of a date object to something pretty
   */
  getDate(d: any) {
    let date = d.date.toDate();
    //console.log(d.date, date);
    return (date.getMonth() + 1) + "/" + date.getDate() + "/" + date.getFullYear();
  }


  /**
   * loads story to be edited into the modal
   * @param element The story to be loaded into modal
   */
  editStory(element: any) {
    this.editedStoryOrig = element;
    this.editing = true;
    //console.log("EL: ",element);
    var date = formatDate(this.startDate, 'yyyy-MM-dd', 'en');
    //console.log("date: ", date);
    this.createPostForm = this.fb.group({
      'postTitle': [element.title, Validators.required],
      'postDate': [date],
      'postSlug': [element.slug],
      'postTags': [element.tags.join(', ')],
      'postImg': [null],
      'editor': new FormControl(element.body),
      'postOnPublic': [element.postPublic],
      'postOnMembers': [element.postMembers]
    });


    var sub = this.storiesService.getImg(element.img).subscribe((url) => {
      this.prevImgURL = url;
      sub.unsubscribe();
    });
  }

  /**
   * deletes the corresponding story document from firebase
   */
  delStory(element: any) {

    if (confirm("Do you want to delete '" + element.title + "'?") == true) {
      this.storiesService.deleteStory(element.id).then(() => {
        //console.log("deleted!");
      });
    }

  }


  get createPostFormControl() {
    return this.createPostForm.controls;
  }

}
