import { Component, ElementRef, OnDestroy, OnInit, ViewChild } from '@angular/core';
import { AngularFireStorage } from '@angular/fire/storage';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatPaginator } from '@angular/material/paginator';
import { MatTableDataSource } from '@angular/material/table';
import { Subscription } from 'rxjs';
import { Measure } from 'src/app/interfaces/measure';
import { MeasuresService } from 'src/app/services/measures.service';

@Component({
  selector: 'app-archive-management',
  templateUrl: './archive-management.component.html',
  styleUrls: ['./archive-management.component.scss']
})
export class ArchiveManagementComponent implements OnInit, OnDestroy {

  createMeasureForm!: FormGroup;

  uploadProgress: number = 0.0;

  public uploading: boolean = false;
  public submitted: boolean = false;
  public editing: boolean = false;

  private measureSub?: Subscription;
  public measures: Measure[] = [];
  public editedMeasureOrig: any = null;

  public newArticleFiles: FileList | undefined;
  public newCodebookFiles: FileList | undefined;
  public newPsychometricsFiles: FileList | undefined;
  public newScalesFiles: FileList | undefined;

  public currentArticleURLs: string[] = [];
  public currentCodebookURLs: string[] = [];
  public currentPsychometricsURLs: string[] = [];
  public currentScalesURLs: string[] = [];

  currentMeasures!: MatTableDataSource<Measure>;
  displayedColumns: string[] = ['title', 'summary', 'authors', 'tags', 'edit'];

  @ViewChild(MatPaginator) paginator!: MatPaginator;
  @ViewChild('createModal') createModal!: ElementRef;
  @ViewChild('articleFileUpload') articleFileUpload!: ElementRef;
  @ViewChild('codebooksFileUpload') codebooksFileUpload!: ElementRef;
  @ViewChild('psychometricsFileUpload') psychometricsFileUpload!: ElementRef;
  @ViewChild('scalesFileUpload') scalesFileUpload!: ElementRef;

  editorOptions = {
    toolbar: {
      container: [
        ["bold", "italic", "underline"],
        [{ 'list': 'ordered' }, { 'list': 'bullet' }],
        [{ 'indent': '-1' }, { 'indent': '+1' }],
        [{ 'color': [] }],
        [{ 'align': [] }],
        ['clean']
      ]
    }
  };
  placeholderText: string = "Write something here...";

  constructor(private fb: FormBuilder, private measuresService: MeasuresService, private afStorage: AngularFireStorage) { }

  /**
   * Initializes the reactive form, and sets up the subscription to get all measures for the measures table.
   */
  ngOnInit(): void {

    this.createMeasureForm = this.fb.group({
      'measureTitle': ['', Validators.required],
      'measureSummary': ['', Validators.required],
      'measureAuthors': [''],
      'measureTags': [''],
      'measureArticles': [''],
      'measureCodebooks': [''],
      'measurePsychometrics': [''],
      'measureScales': ['']
    });

    this.measureSub = this.measuresService.getMeasures().subscribe((measures => {
      this.measures = measures as Measure[];
      this.currentMeasures = new MatTableDataSource<Measure>(this.measures);
      this.currentMeasures.paginator = this.paginator;
      //console.log(this.measures);
    }));
  }

  /**
   * Destroys the measures service subscription.
   */
  ngOnDestroy() {
    this.measureSub?.unsubscribe();
  }

  /**
   * Entry point for submitting the measure modal form, logic is different depending on if editing or creating.
   */
  onSubmit() {
    if (this.editing) this.onMeasureEdit();
    else this.onMeasureCreate();
  }

  /**
   * Called when the new measure modal pops up, used to clear old data from the form.
   */
  onAdd() {
    if (this.editing) {
      this.resetForm();
    }
  }

  /**
   * Fills the edit measure modal form with data from the measure clicked on from the measure table and stores relevant clean data just in case the operation is cancelled.
   * 
   * @param element The measure data to fill out the modal form with.
   */
  editMeasure(element: any) {
    this.editedMeasureOrig = element;
    this.editing = true;

    this.createMeasureForm = this.fb.group({
      'measureTitle': [element.title, Validators.required],
      'measureSummary': [element.summary, Validators.required],
      'measureAuthors': [element.authors.join(', ')],
      'measureTags': [element.tags.join(', ')],
      'measureArticles': [element.articles.join(', ')],
      'measureCodebooks': [element.codebooks.join(', ')],
      'measurePsychometrics': [element.psychometrics.join(', ')],
      'measureScales': [element.scales.join(', ')]
    });

    this.currentArticleURLs = element.articles;
    this.currentCodebookURLs = element.codebooks;
    this.currentPsychometricsURLs = element.psychometrics;
    this.currentScalesURLs = element.scales;
  }

  /**
   * The main function for creating a new measure in the database. 
   * 
   * Gets all of the files from the file uploads, puts a new measure in the measures document store in the DB, then sequentially uploads all
   * of the files necessary to the file store in the DB. If any of this fails, everything is undone. 
   */
  onMeasureCreate() {

    var articles = [];
    var codebooks = [];
    var psychometrics = [];
    var scales = [];

    var files: any[] = [];
    var numFilesUploaded = 0;
    this.uploadProgress = 0;

    this.uploading = true;

    if (this.newArticleFiles) {
      for (var i = 0; i < this.newArticleFiles.length; i++) {
        files.push(this.newArticleFiles[i]);
        articles.push(this.newArticleFiles[i].name);
      }
    }

    if (this.newCodebookFiles) {
      for (var i = 0; i < this.newCodebookFiles.length; i++) {
        files.push(this.newCodebookFiles[i]);
        codebooks.push(this.newCodebookFiles[i].name);
      }
    }

    if (this.newPsychometricsFiles) {
      for (var i = 0; i < this.newPsychometricsFiles.length; i++) {
        files.push(this.newPsychometricsFiles[i]);
        psychometrics.push(this.newPsychometricsFiles[i].name);
      }
    }

    if (this.newScalesFiles) {
      for (var i = 0; i < this.newScalesFiles.length; i++) {
        files.push(this.newScalesFiles[i]);
        scales.push(this.newScalesFiles[i].name);
      }
    }

    this.measuresService.createMeasure({
      title: this.createMeasureForm.get('measureTitle')?.value,
      summary: this.createMeasureForm.get('measureSummary')?.value,
      authors: this.parseCommaSeparatedList(this.createMeasureForm.get('measureAuthors')?.value),
      tags: this.parseCommaSeparatedList(this.createMeasureForm.get('measureTags')?.value),
      articles: articles,
      codebooks: codebooks,
      psychometrics: psychometrics,
      scales: scales
    }).then((docRef) => {

      //no files to upload
      if (files.length == 0) {
        this.createModal.nativeElement.click();
        this.resetForm();
      }

      for (var i = 0; i < files.length; i++) {

        //console.log("starting upload task!")
        //console.log(files[i]);

        var path = docRef.id + '/' + files[i].name;
        let task = this.afStorage.upload(path, files[i]);
        task.then((a) => {

          //console.log("upload task complete! ", a);
          numFilesUploaded++;
          this.uploadProgress = numFilesUploaded / files.length;

          if (numFilesUploaded == files.length) {
            this.createModal.nativeElement.click();
            this.resetForm();
          }

        }, (error) => {
          console.error('Failed to upload file with error: ', error);
          docRef.delete();
        });
      }
    });
  }

  /**
   * The main function for editing an existing measure in the database. 
   * 
   * Gets all of the files from the file uploads, puts a new measure in the measures document store in the DB, then sequentially uploads all
   * of the files necessary to the file store in the DB. If any of this fails, everything is undone. This also supports deleting existing files
   * by removing them from the document and deleting them from the file store. 
   */
  onMeasureEdit() {
    if (this.createMeasureForm.pristine) {
      this.createModal.nativeElement.click();
    } else {
      var articles = this.currentArticleURLs;
      var codebooks = this.currentCodebookURLs;
      var psychometrics = this.currentPsychometricsURLs;
      var scales = this.currentScalesURLs;

      var files: any[] = [];
      var numFilesUploaded = 0;
      this.uploadProgress = 0.0;

      this.uploading = true;

      if (this.newArticleFiles) {
        for (var i = 0; i < this.newArticleFiles.length; i++) {
          files.push(this.newArticleFiles[i]);
          articles.push(this.newArticleFiles[i].name);
        }
      }

      if (this.newCodebookFiles) {
        for (var i = 0; i < this.newCodebookFiles.length; i++) {
          files.push(this.newCodebookFiles[i]);
          codebooks.push(this.newCodebookFiles[i].name);
        }
      }

      if (this.newPsychometricsFiles) {
        for (var i = 0; i < this.newPsychometricsFiles.length; i++) {
          files.push(this.newPsychometricsFiles[i]);
          psychometrics.push(this.newPsychometricsFiles[i].name);
        }
      }

      if (this.newScalesFiles) {
        for (var i = 0; i < this.newScalesFiles.length; i++) {
          files.push(this.newScalesFiles[i]);
          scales.push(this.newScalesFiles[i].name);
        }
      }

      this.measuresService.updateMeasure(this.editedMeasureOrig.id, {
        title: this.createMeasureForm.get('measureTitle')?.value,
        summary: this.createMeasureForm.get('measureSummary')?.value,
        authors: this.parseCommaSeparatedList(this.createMeasureForm.get('measureAuthors')?.value),
        tags: this.parseCommaSeparatedList(this.createMeasureForm.get('measureTags')?.value),
        articles: articles,
        codebooks: codebooks,
        psychometrics: psychometrics,
        scales: scales
      }).then(() => {

        //use numFiles to calculate task bar percentage

        //no files to upload
        if (files.length == 0) {
          this.createModal.nativeElement.click();
          this.resetForm();
        }

        for (var i = 0; i < files.length; i++) {

          //console.log("starting upload task!");
          //console.log(files[i]);

          var path = this.editedMeasureOrig.id + '/' + files[i].name;
          let task = this.afStorage.upload(path, files[i]);
          task.then((a) => {

            //console.log("upload task complete! ", a);
            numFilesUploaded++;
            this.uploadProgress = numFilesUploaded / files.length;

            if (numFilesUploaded == files.length) {
              this.createModal.nativeElement.click();
              this.resetForm();
            }

          }, (error) => {
            console.error('Failed to upload file with error: ', error);

            //Here is a big boy decision I made that could've been handled more
            //elegantly. If any of the files fail to upload, I keep all of the rest of
            //the updated fields but revert the entire file links back to what they were pre-edit.
            //the only issue i see with this is that there may have been some files that successfully
            //uploaded but those get unlinked. otherwise, this solution works well i think!
            this.measuresService.updateMeasure(this.editedMeasureOrig.id, {
              title: this.createMeasureForm.get('measureTitle')?.value,
              summary: this.createMeasureForm.get('measureSummary')?.value,
              authors: this.parseCommaSeparatedList(this.createMeasureForm.get('measureAuthors')?.value),
              tags: this.parseCommaSeparatedList(this.createMeasureForm.get('measureTags')?.value),
              articles: this.currentArticleURLs,
              codebooks: this.currentCodebookURLs,
              psychometrics: this.currentPsychometricsURLs,
              scales: this.currentScalesURLs
            });

          });
        }
      });
    }
  }

  /**
   * Resets the state of the measure editing modal, clears out the file lists and the file upload elements.  
   */
  resetForm() {

    this.uploadProgress = 0.0;
    this.uploading = false;
    this.editing = false;
    this.editedMeasureOrig = null;

    this.createMeasureForm = this.fb.group({
      'measureTitle': ['', Validators.required],
      'measureSummary': ['', Validators.required],
      'measureAuthors': [''],
      'measureTags': [''],
      'measureArticles': [''],
      'measureCodebooks': [''],
      'measurePsychometrics': [''],
      'measureScales': ['']
    });

    this.newArticleFiles = undefined;
    this.newCodebookFiles = undefined;
    this.newPsychometricsFiles = undefined;
    this.newScalesFiles = undefined;

    this.articleFileUpload.nativeElement.value = '';
    this.codebooksFileUpload.nativeElement.value = '';
    this.psychometricsFileUpload.nativeElement.value = '';
    this.scalesFileUpload.nativeElement.value = '';

    this.currentArticleURLs = [];
    this.currentCodebookURLs = [];
    this.currentPsychometricsURLs = [];
    this.currentScalesURLs = [];
  }

  /**
   * Deletes a measure from the measures document store and removes all associated files from the file store.
   * 
   * @param element The measure to delete. 
   */
  deleteMeasure(element: any) {
    if (confirm("Do you want to delete '" + element.title + "'?") == true) {
      this.uploading = true;
      this.measuresService.deleteMeasure(element.id).then(() => {
        element.articles.forEach((name: any) => {
          this.afStorage.ref(element.id + '/' + name).delete();
        });
        element.codebooks.forEach((name: any) => {
          this.afStorage.ref(element.id + '/' + name).delete();
        });
        element.psychometrics.forEach((name: any) => {
          this.afStorage.ref(element.id + '/' + name).delete();
        });
        element.scales.forEach((name: any) => {
          this.afStorage.ref(element.id + '/' + name).delete();
        });
        this.uploading = false;
      });
    }
  }

  /**
   * Deletes a file from the file store with prompt, and then removes it from the file arrays to be saved. 
   * 
   * @param category The category of file being deleted, in order to maintain records in the measure document. 
   * @param filename The name of the file to be deleted
   */
  deleteFile(category: string, filename: string) {
    if (confirm("Do you want to delete '" + filename + "'? This action cannot be undone.") == true) {
      this.uploading = true;
      this.afStorage.ref(this.editedMeasureOrig.id + "/" + filename).delete().toPromise().then((data) => {
        //console.log("successfully deleted file");
        switch (category) {
          case 'article':
            var index = this.currentArticleURLs.indexOf(filename);
            this.currentArticleURLs.splice(index, 1);
            break;
          case 'codebook':
            var index = this.currentCodebookURLs.indexOf(filename);
            this.currentCodebookURLs.splice(index, 1);
            break;
          case 'psycho':
            var index = this.currentPsychometricsURLs.indexOf(filename);
            this.currentPsychometricsURLs.splice(index, 1);
            break;
          case 'scales':
            var index = this.currentScalesURLs.indexOf(filename);
            this.currentScalesURLs.splice(index, 1);
            break
        }

        this.measuresService.updateMeasure(this.editedMeasureOrig.id, {
          title: this.editedMeasureOrig.title,
          summary: this.editedMeasureOrig.summary,
          authors: this.editedMeasureOrig.authors,
          tags: this.editedMeasureOrig.tags,
          articles: this.currentArticleURLs,
          codebooks: this.currentCodebookURLs,
          psychometrics: this.currentPsychometricsURLs,
          scales: this.currentScalesURLs
        }).then(() => {
          this.uploading = false;
        })
      }, (error) => {
        console.error("failed to delete file: ", error);
      });
    }
  }

  /**
   * Returns the controls object from the reactive form.
   */
  get createMeasureFormControl() {
    return this.createMeasureForm.controls;
  }

  /**
   * Change detection tracker for the file lists.
   * 
   * @param category The category of the files, so we know which array to update.
   * @param fileEvent The HTML event that occurs when files are uploaded. 
   */
  public onChangeFiles(category: string, fileEvent: any) {

    var newFiles = fileEvent.target.files;

    switch (category) {
      case 'article':
        this.newArticleFiles = newFiles;
        break;
      case 'codebook':
        this.newCodebookFiles = newFiles;
        break;
      case 'psycho':
        this.newPsychometricsFiles = newFiles;
        break;
      case 'scales':
        this.newScalesFiles = newFiles;
        break
    }
  }


  /**
   * Helper function to convert an array to a comma and space separated string.
   * 
   * @param array List of stringable objects
   * @returns A comma separated string of the array elements. 
   */
  parseArray(array: any[]) {
    var arrayAsString = '';

    array.forEach((element: any, index: number) => {
      arrayAsString += element;
      if (index < array.length - 1) {
        arrayAsString += ', ';
      }
    });

    return arrayAsString;
  }

  /**
   * Converts a string with commas to an array
   * 
   * @param text The string to convert to array
   * @returns The array with the elements split by comma.
   */
  parseCommaSeparatedList(text: String) {
    let r = RegExp(" *, *");
    let a = text.split(r);
    return a;
  }

}
