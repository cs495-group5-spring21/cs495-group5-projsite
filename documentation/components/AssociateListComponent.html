import { Component, OnInit } from '@angular/core';
import { Associate } from 'src/app/interfaces/associate';
import { PeopleService, PeopleTypes } from 'src/app/services/people.service';

@Component({
  selector: 'app-associate-list',
  templateUrl: './associate-list.component.html',
  styleUrls: ['./associate-list.component.scss']
})
export class AssociateListComponent implements OnInit {

  associates: Associate[] = [];

  constructor(private peopleService: PeopleService) { }

  /**
   * Loads the associates array with each document from the associates collection in the database to be displayed on the Associates page of the website. 
   */
  ngOnInit(): void {
    this.peopleService.getPeople<Associate>(PeopleTypes.ASSOCIATES).subscribe(data => this.associates = data);
  }

}
