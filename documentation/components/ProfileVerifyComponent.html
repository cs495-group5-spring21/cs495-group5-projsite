import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { FirebaseAuthService } from 'src/app/services/firebase-auth.service';

@Component({
  selector: 'app-profile-verify',
  templateUrl: './profile-verify.component.html',
  styleUrls: ['./profile-verify.component.scss']
})
export class ProfileVerifyComponent implements OnInit {

  public profileForm!: FormGroup;
  public formError: boolean = false; //might need this ?
  submitted = false;
  public formSubmit = false;

  constructor(
    private fireBaseAuthService: FirebaseAuthService,
    private fb: FormBuilder,
    public router: Router
  ) { }

  /**
   * Initializes a profile form group with the necessary variables for the user's profile to be completed and adds validators to sections that require information.
   */
  ngOnInit(): void {
    this.profileForm = this.fb.group({
      firstName: ['', Validators.required],
      lastName: ['', Validators.required],
      association: ['', Validators.required],
      userState: ['', Validators.required],
      userAreaofStudy: ['', Validators.required],
      userCity: [''],
      userZip: ['']
    });
  }   //end of ngOnInit()


  /**
   * On the submission of the form, checks the validators to approve the submission, 
   * updates user info, and routes the user to the archive.
   */
  public onFormSubmit() { //makes sure all validators are correct 
    this.formError = false;
    this.submitted = true;

    if (this.profileForm.valid) {
      const firstName = this.profileForm.get('firstName')?.value;
      const lastName = this.profileForm.get('lastName')?.value;
      const association = this.profileForm.get('association')?.value;
      const userState = this.profileForm.get('userState')?.value;
      const userAreaofStudy = this.profileForm.get('userAreaofStudy')?.value;
      const userCity = this.profileForm.get('userCity')?.value;
      const userZip = this.profileForm.get('userZip')?.value;

      this.fireBaseAuthService.UpdateUserDocData({
        'firstName': firstName,
        'lastName': lastName,
        'association': association,
        'userState': userState,
        'userCity': userCity,
        'userAreaofStudy': userAreaofStudy,
        'userZip': userZip,
        'initialized': true
      })

      //console.log('Profile was Validated');
    } else () => {
      this.formSubmit = true;
      //var userJson =  localStorage.getItem('user');
      //console.log(JSON.stringify(this.userData));
    }
    this.router.navigate(['archive']);
  }

  /**
   * Gets the profileForm controls as to allow the use of them throughout this typescript code.
   */
  get profileFormControl() {
    return this.profileForm.controls;
  }

} //end of ProfileVerify class
