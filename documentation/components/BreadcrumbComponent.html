import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';

@Component({
  selector: 'app-breadcrumb',
  templateUrl: './breadcrumb.component.html',
  styleUrls: ['./breadcrumb.component.scss']
})
export class BreadcrumbComponent implements OnInit {

  public breadcrumbs: any[] = [];
  public parent: any = '';
  public parentUrl: any = '';
  public isDynamic = false;
  public dynamicBreadcrumb: any = '';
  
  constructor(private router: Router, private activatedRoute: ActivatedRoute) { }

  ngOnInit(): void {
    if(this.activatedRoute.snapshot && this.activatedRoute.snapshot.data && this.activatedRoute.snapshot.data.breadcrumbs){
      this.parent = this.activatedRoute.snapshot.data.parent;
      this.parentUrl = this.activatedRoute.snapshot.data.parentUrl;
      this.breadcrumbs = this.activatedRoute.snapshot.data.breadcrumbs;
      
      if(this.activatedRoute.snapshot.data.isDynamic){
        this.isDynamic = this.activatedRoute.snapshot.data.isDynamic;
        this.activatedRoute.paramMap.subscribe(params => {
          this.dynamicBreadcrumb = params.get(this.breadcrumbs[0])!;
        });
      }

    }
  }

}
