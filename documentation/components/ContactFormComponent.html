import { Component, OnInit } from '@angular/core';
import { AngularFirestore, DocumentReference } from '@angular/fire/firestore';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';


@Component({
  selector: 'app-contact-form',
  templateUrl: './contact-form.component.html',
  styleUrls: ['./contact-form.component.scss']
})
export class ContactFormComponent implements OnInit {

  contactForm !: FormGroup;
  public sending: boolean = false;
  public error: boolean = false;

  constructor(private db: AngularFirestore, private fb: FormBuilder) { }

  ngOnInit(): void {
    this.resetForm();
  }

  /**
   * sends contact info to firebase
   * @param data the contact to be added
   * @returns the promise to the action
   */
  createContact(data: any): Promise<DocumentReference<any>> {
    return this.db.collection('contact').add(data);
  }

  resetForm() {
    this.contactForm = this.fb.group({
      'name': ['', Validators.required],
      'email': ['', Validators.email], //FIXME VALIDATE EMAIL
      'message': ['', Validators.required]
    })
  }

  /**
   * creates the contact form and calls createContact()
   */
  onSubmit() {
    //console.log(this.contactForm);
    if (this.contactForm.status == "INVALID") {
      this.error = true;
    }
    else {
      this.error = false;
      this.createContact({
        date: new Date(),
        name: this.contactForm.get('name')?.value,
        email: this.contactForm.get('email')?.value,
        message: this.contactForm.get('message')?.value,
      }).then(() => {
       // console.log("Submitted!");
        //todo: add success popup
        this.resetForm();
        this.displayMessage();
      })
    }
   
  }

  /**
   * displays success message for 5 seconds 
   */
  displayMessage() {
    this.sending = true;
    setTimeout(() => { this.sending = false; }, 5000);
  }

}
