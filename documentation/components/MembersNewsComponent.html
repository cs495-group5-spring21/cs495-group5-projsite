import { Component, OnInit } from '@angular/core';
import { Story } from 'src/app/interfaces/story';
import { StoriesService } from 'src/app/services/stories.service';
import { DateFormatter } from '../../models/DateFormatter';

@Component({
  selector: 'app-news',
  templateUrl: './news.component.html',
  styleUrls: ['./news.component.scss']
})
export class MembersNewsComponent implements OnInit {

  public stories: Story[] = [];

  constructor(private storiesService: StoriesService) { }

  ngOnInit(): void {
    this.stories = this.storiesService.getStoriesMembers();

    //console.log(this.stories);
  }

  /**
   * formats and prints first two tags
   * @returns formatted tag string
   */
  getTags(story: Story): string {
    let output: string = "";
    output += story.tags[0];
    if (story.tags.length > 1) output += ", " + story.tags[1];
    return output;
  }

  getDate(story: Story) {
    return DateFormatter.getDate(story);
  }

}
