import { Component, OnInit } from '@angular/core';
import { Observable, Subscription } from 'rxjs';
import { Measure } from 'src/app/interfaces/measure';
import { MeasuresService } from 'src/app/services/measures.service';

@Component({
  selector: 'app-measures',
  templateUrl: './measures.component.html',
  styleUrls: ['./measures.component.scss']
})
export class MeasuresComponent implements OnInit {

  constructor(private measuresService: MeasuresService) { }

  private allMeasuresObs!: Observable<Measure[]>;
  private measureSub?: Subscription;

  //all measures at all times
  public allMeasures: Measure[] = [];

  //all measures (including those not shown) but all filtered
  public filteredMeasures: Measure[] = [];

  //currently shown measures
  public currentMeasures: Measure[] = [];

  public displayCount: number = 4;
  public pageNumber: number = 0;
  public numPages: number = 0;
  public pagination: (number)[] = [];

  public tags: any[] = [];
  public filterTags: any[] = [];
  public filter!: string;

  ngOnInit(): void {
    this.allMeasuresObs = this.measuresService.getMeasures();

    this.measureSub = this.allMeasuresObs.subscribe((measures => {
      this.allMeasures = measures as Measure[];
      console.log(this.allMeasures);

      this.numPages = this.allMeasures.length / this.displayCount;
      this.numPages = Math.ceil(this.numPages);

      this.tags = this.measuresService.getTags();

      //console.log("num measures: " + this.allMeasures.length + ", current page: " + this.pageNumber + ", num pages: " + this.numPages);
      this.getCurrentPageMeasures();
      this.calculatePagination();
    }));
  }

  ngOnDestroy() {
    this.measureSub?.unsubscribe();
  }

  /**
   * calculates array of numbers for pagination display, adds it to pagination array
   */
  calculatePagination() {
    this.pagination = [];
    if (this.numPages > 10) { //truncate
      if (this.pageNumber > 4 || this.pageNumber < this.numPages - 5) {
        for (let i = this.pageNumber - 4; i < this.pageNumber + 5; ++i) {
          this.pagination.push(i);
        }
      }
      else {
        this.pagination.concat([1, 2, 3, 4, 5, -1]);
        for (let i = this.numPages - 5; i < this.numPages; ++i) this.pagination.push(i + 1);
      }
    }
    else for (let i = 1; i <= this.numPages; ++i) this.pagination.push(i);

    //console.log(this.pagination);
  }


  getCurrentPageMeasures() {
    this.onFilterSearch();
    this.currentMeasures = this.filteredMeasures.slice(this.pageNumber * this.displayCount, this.pageNumber * this.displayCount + this.displayCount);
    //console.log("post getCurrentPageMeasures() currentMeasures: ");
    //console.log(this.currentMeasures);
  }

  onFilterSearch() {

    this.filteredMeasures = this.allMeasures;

    if (this.filterTags.length > 0) {
      this.filteredMeasures = this.filteredMeasures.filter((measure) => {
        if (measure.tags) {
          return measure.tags.some(t => this.filterTags.indexOf(t) !== -1)
        }

        return false;
      });

      //console.log("filtered measures after tags filter:");
      //console.log(this.filteredMeasures);
    }

    if (this.filter && this.filter != '') {
      this.filteredMeasures = this.filteredMeasures.filter((measure) => JSON.stringify(measure).toLowerCase().indexOf(this.filter.toLowerCase()) !== -1);
    }

    this.numPages = this.filteredMeasures.length / this.displayCount;
    this.numPages = Math.ceil(this.numPages);
    this.calculatePagination();

  }

  onFilterReset() {
    this.filter = '';
    this.filterTags = [];

    //unhighlight tags
    let elements = document.querySelectorAll(' .pop-tags');
    elements.forEach((element) => {
      element.classList.remove('tag-button-active');
    });

    this.getCurrentPageMeasures();
    this.goToPage(0);
    this.scrollToTop(); //fixme do we like scroll?
  }


  onTagClick($event: any, tag: string) {
    var tagIndex = this.filterTags.indexOf(tag);
    //console.log(tagIndex);

    $event.target.classList.toggle('tag-button-active');

    if (tagIndex !== -1) {
      this.filterTags = this.filterTags.filter(el => el != tag);
    } else {
      this.filterTags.push(tag);
    }

    this.getCurrentPageMeasures()
    this.goToPage(0);
  }

  nextPage() {

    this.pageNumber++;

    if (this.pageNumber >= this.numPages) {
      this.pageNumber--;
    }

    this.getCurrentPageMeasures();
    this.calculatePagination();
  }

  previousPage() {
    this.pageNumber--;

    if (this.pageNumber < 0) {
      this.pageNumber++;
    }
    this.getCurrentPageMeasures();
    this.calculatePagination();
  }

  goToPage(page: number) {
    this.pageNumber = page;

    if (this.pageNumber < 0) {
      this.pageNumber++;
    }
    if (this.pageNumber >= this.numPages) {
      this.pageNumber--;
    }

    this.getCurrentPageMeasures();
    this.calculatePagination();
  }

  onChangeResultsPerPage(count: number) {
    this.displayCount = count;
    this.goToPage(0);
  }

  /**
   * resets pagination when search button is pressed
   */
  onSearch() {
    this.getCurrentPageMeasures();
    this.goToPage(0);
    this.scrollToTop(); 
  }

  scrollToTop() {
    let el = document.getElementById('top');
    if (el) {
      el.scrollIntoView({ behavior: "smooth", block: "end" });
    }
  }

}
