import { Component, OnInit, OnDestroy } from '@angular/core';
import { Router } from '@angular/router';
import { FirebaseAuthService } from 'src/app/services/firebase-auth.service';

@Component({
  selector: 'app-verify-email',
  templateUrl: './verify-email.component.html',
  styleUrls: ['./verify-email.component.scss']
})
export class VerifyEmailComponent implements OnInit, OnDestroy {

  emailSent: boolean = false;
  verify: boolean = false;

  timerHandle: any;

  constructor(private firebaseAuthService: FirebaseAuthService, private router: Router) { }

  ngOnInit(): void {
    //TODO: this doesn't work because the timer doesn't have access to the firebase service (scope changes from outside of this component.). can look into fixes if desired. 
    //this.timerHandle = setInterval(this.checkVerify, 10000);
  }

  ngOnDestroy(): void {
    //clearInterval(this.timerHandle);
  }

  /**
   * Sends the user currently registering an email verification email to autheticate their account. 
   */
  sendEmail() {
    this.firebaseAuthService.SendVerificationMail();
    this.emailSent = true;
  }

  /**
   * Calls the checkVerify function on the click of the "Check Vrification" button.
   */
  checkVerification() {
    this.checkVerify();
  }

  /**
   * In this method, the user's firebase information is reloaded and the user's local storage is updated to check if their email has been verified.
   */
  checkVerify() {
    this.firebaseAuthService.GetAuthUserState().then((user) => {
      //console.log("user state in checkVerification(): ", user);
      if (user && user !== null) {
        user.reload().then(() => {
          this.firebaseAuthService.GetAuthUserState().then((reloadedUser) => {
            if (user && user !== null) {
              this.firebaseAuthService.UpdateLocalStorageUser(user);
              if (this.firebaseAuthService.userData && this.firebaseAuthService.userData.emailVerified) {
                this.router.navigate(['archive']);
              }
            }
          });
        });
      }
    });
  }
}
