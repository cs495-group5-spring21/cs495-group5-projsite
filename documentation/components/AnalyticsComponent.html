On the initialization of the page, there are multiple things done.

  1. The userSignIn event documents are loaded into an array, which then has analytics done on it to display the most recent user sign ins. That information is then formatted and displayed on the analytics web page.

  2. The measureView event documents, as well as all of the measure documents, are loaded into separate arrays. They are then crossreferenced to find the measureViews on existing documents. The views are then counted, total view counts for each measurement are stored in a map and then that map is used to find the top 3 and the bottom 3 viewed measures. The top viewed measure is displayed on a graph to show views over time, which uses each of the top measurement's measureView event document timestamps to separate each view over the last four weeks from the day you are accessing the website.

import { MatTableDataSource } from '@angular/material/table';
import { MeasuresService } from './../../services/measures.service';
import { Component, OnInit } from '@angular/core';
import { AnalyticsService } from 'src/app/services/analytics.service';
import { Analytic } from 'src/app/interfaces/analytic';
import { UserService } from 'src/app/services/user.service';
import { Measure } from 'src/app/interfaces/measure';


@Component({
  selector: 'app-analytics',
  templateUrl: './analytics.component.html',
  styleUrls: ['./analytics.component.scss']
})
export class AnalyticsComponent implements OnInit {

  mostViewedMeasureName: string = '';
  mostViewedGraphData: any[] = [];

  graphSize: [number, number] = [700, 300];
  xAxisLabel: string = 'Date';
  yAxisLabel: string = 'Views';

  userDataSource!: MatTableDataSource<any>;
  mostViewedMeasuresDataSource!: MatTableDataSource<any>;
  leastViewedMeasuresDataSource!: MatTableDataSource<any>;

  measuresColumns: string[] = ['measureTitle', 'viewCount'];
  userColumns: string[] = ['name', 'time'];

  signIns: Analytic[] = [];
  measureViews: Analytic[] = [];
  allMeasures: Measure[] = [];

  userNames: Map<string, any> = new Map();
  mostViewed: Map<string, any> = new Map();   //will hold top three (different) most viewed (high to low)
  leastViewed: Map<string, any> = new Map();   //will hold bottom three least viewed (high to low)
  perDocViews: Map<string, any> = new Map();      //new map to hold the amount of views per document
  finalDocViews: Map<string, number> = new Map();

  //GOTTA MAKE THE OBJECT AND USE IT TO COMBINE THE INFOMRATION FROM THE MEASURES COLLECTION 
  //AND THE ANALYTICS COLLECTION BASE ON THEIR MEASRUE IDS


  constructor(public measuresService: MeasuresService, public analyticsService: AnalyticsService, public userService: UserService) { }

  /**
   * Keeps the order of the map objects when displaying them on the webpage.
   * @param a The index of the map where the object comes before b.
   * @param b The index of the map where the object comes after a. 
   * @returns 1 to ensure the order stays as set. 
   */
  asIsOrder(a: any, b: any) {
    return 1;
  }

  /**
   * On the initialization of the page, there are multiple things done.
   * 1. The userSignIn event documents are loaded into an array, which then has analytics done on it to display the most recent user sign ins. 
   * That information is then formatted and displayed on the analytics web page.
   * 
   * 2. The measureView event documents, as well as all of the measure documents, are loaded into separate arrays. They are then crossreferenced to find the measureViews on existing documents.
   * The views are then counted, total view counts for each measurement are stored in a map and then that map is used to find the top 3 and the bottom 3 viewed measures. 
   * The top viewed measure is displayed on a graph to show views over time, which uses each of the top measurement's measureView event document timestamps to separate each view over 
   * the last four weeks from the day you are accessing the website. 
   */
  ngOnInit(): void {

    this.analyticsService.getAnalytics<Analytic>("userSignIn").subscribe((data) => {
      this.signIns = data
      var userCount = 0;

      this.signIns.sort((a, b) => b.timestamp - a.timestamp);

      this.signIns.forEach((signIn) => {
        if (!this.userNames.has(signIn.userId)) {
          this.userNames.set(signIn.userId, {
            name: undefined,
            time: new Date(signIn.timestamp),
            id: signIn.userId,
          });
        }
      });

      this.userNames.forEach((userData, userId) => {
        this.userService.getUserData(userId).subscribe((d) => {
          if (d?.firstName != undefined) {
            this.userNames.set(userId, {
              name: d.firstName,
              time: userData.time,
              id: userId,
            });
            if (++userCount == this.userNames.size) {
              this.userDataSource = new MatTableDataSource<any>(Array.from(this.userNames.values()));
            }
          }
        });
      });
    });

    this.analyticsService.getAnalytics<Analytic>("measureViewed").subscribe((data) => {
      this.measuresService.getMeasures().subscribe((allMeasures) => {
        this.measureViews = data;    //all of the measureView event documents
        this.allMeasures = allMeasures; //all of the existing measures in the database
        this.perDocViews.clear();   //clears it to prevent duplicates if multiple subscriptions are called

        this.allMeasures.forEach((measure) => {
          //Key: measureId Value: Measure Title, Count
          if (!this.perDocViews.has(measure.id)) {
            this.perDocViews.set(measure.id, {
              measureTitle: measure.title,
              viewCount: 0,
              timestamp: 0,
              id: measure.id
            });
          }
          //console.log("Measure added: ", this.perDocViews.keys(), " with its data: ", this.perDocViews.get(measure.id));
        });

        this.measureViews.forEach((measureView) => {
          if (this.perDocViews.has(measureView.measureId)) {
            var value = this.perDocViews.get(measureView.measureId);
            value.viewCount++;
            value.timestamp = measureView.timestamp;
            this.perDocViews.set(measureView.measureId, value);
          }
        });

        let sortedDocs = new Map([...this.perDocViews].sort(([k, v], [k2, v2]) => {
          if (v.viewCount < v2.viewCount) {
            return 1;
          }
          if (v.viewCount > v2.viewCount) {
            return -1;
          }
          return 0;
        }));

        //console.log("The sorted map: ", sortedDocs);

        this.mostViewed.clear();
        var i = 0;
        sortedDocs.forEach((value: any, key: string) => {
          if (!this.mostViewed.has(key) && i < 3) {
            this.mostViewed.set(key, value);
            i++;
          } //end of if statement
        });

        var mostViewedArr = Array.from(this.mostViewed.values());
        this.mostViewedMeasuresDataSource = new MatTableDataSource<any>(mostViewedArr);
        //console.log("The mostviewed map: ", this.mostViewed);      

        this.leastViewed.clear();
        i = 0;
        sortedDocs.forEach((value: any, key: string) => {
          if (!this.leastViewed.has(key) && i >= (sortedDocs.size - 3)) {
            this.leastViewed.set(key, value);
            i++;
          } //end of if statement
          //if the amount of documents is only a size of 3
          else if (sortedDocs.size <= 3) {
            this.leastViewed.set(key, value);
          }
          i++;
        });

        this.leastViewedMeasuresDataSource = new MatTableDataSource<any>(Array.from(this.leastViewed.values()));
        //console.log("The leastviewed map: ", this.leastViewed);  

        var measureToGraph = 0;
        this.mostViewedMeasureName = mostViewedArr[measureToGraph].measureTitle;
        var graphObject = {
          "name": mostViewedArr[measureToGraph].measureTitle,
          "series": [] as any[]
        };
        var graphObjectSeries: any[] = [];

        var currentDate = new Date();
        var now = currentDate.getTime();

        var pastDate = currentDate.getDate() - 7;
        currentDate.setDate(pastDate);
        var oneWeekAgo = currentDate.getTime();

        var pastDate = currentDate.getDate() - 7;
        currentDate.setDate(pastDate);
        var twoWeeksAgo = currentDate.getTime();

        var pastDate = currentDate.getDate() - 7;
        currentDate.setDate(pastDate);
        var threeWeeksAgo = currentDate.getTime();

        var pastDate = currentDate.getDate() - 7;
        currentDate.setDate(pastDate);
        var oneMonthAgo = currentDate.getTime();

        var dateRanges = [oneMonthAgo, threeWeeksAgo, twoWeeksAgo, oneWeekAgo, now];
        var viewCounterOverTime = [0, 0, 0, 0];

        //only doing this for the most viewed measure for now.
        this.measureViews.forEach((measureView) => {
          if (measureView.measureId == mostViewedArr[measureToGraph].id) {
            if (measureView.timestamp > dateRanges[0]) {
              if (measureView.timestamp < dateRanges[1]) {  //between one month ago and three weeks ago
                viewCounterOverTime[0]++;
              } else if (measureView.timestamp < dateRanges[2]) {  //between three weeks ago and two weeks ago
                viewCounterOverTime[1]++;
              } else if (measureView.timestamp < dateRanges[3]) {  //between two weeks ago and one week ago
                viewCounterOverTime[2]++;
              } else {  //between one week ago and now
                viewCounterOverTime[3]++;
              }
            }
          }
        });

        graphObjectSeries.push({
          "name": new Date(threeWeeksAgo).toLocaleString().split(',')[0],
          "value": viewCounterOverTime[0]
        });

        graphObjectSeries.push({
          "name": new Date(twoWeeksAgo).toLocaleString().split(',')[0],
          "value": viewCounterOverTime[1]
        });

        graphObjectSeries.push({
          "name": new Date(oneWeekAgo).toLocaleString().split(',')[0],
          "value": viewCounterOverTime[2]
        });

        graphObjectSeries.push({
          "name": new Date(now).toLocaleString().split(',')[0],
          "value": viewCounterOverTime[3]
        });

        graphObject.series = graphObjectSeries;
        this.mostViewedGraphData = [graphObject];
      }); //end of get all measures
    }); //end of the measureView function

  }

}
