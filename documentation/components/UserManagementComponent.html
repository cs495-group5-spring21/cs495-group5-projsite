import { Component, OnInit } from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';
import { FirebaseUser } from 'src/app/interfaces/firebase-user';
import { FirebaseAuthService } from 'src/app/services/firebase-auth.service';
import { UserService } from 'src/app/services/user.service';

@Component({
  selector: 'app-user-management',
  templateUrl: './user-management.component.html',
  styleUrls: ['./user-management.component.scss']
})
export class UserManagementComponent implements OnInit {

  displayedAdminColumns: string[] = ['name','association','email'];
  displayedNormalColumns: string[] = ['name','association','email', 'edit'];
  adminUsersSource!: MatTableDataSource<FirebaseUser>;
  normalUsersSource!: MatTableDataSource<FirebaseUser>;

  constructor(public userService: UserService, public firebaseAuthServic: FirebaseAuthService) { }

  ngOnInit(): void {
    this.userService.getUsers().subscribe((data) => {
      var normalUsers = Array<FirebaseUser>();
      var adminUsers = Array<FirebaseUser>();

      var users = Array<FirebaseUser>();
      users = data as FirebaseUser[];
      users.forEach((user) => {
        if(user.admin) {
          adminUsers.push(user);
        } else {
          normalUsers.push(user);
        }
      });
      this.normalUsersSource = new MatTableDataSource<FirebaseUser>(normalUsers);
      this.adminUsersSource = new MatTableDataSource<FirebaseUser>(adminUsers);
    });
  }

  /**
   * Function called when an administrator wants to delete a user
   * 
   * @param element object contaiing the data of the user document to be deleted
   */
  onDelete(element: FirebaseUser) {
    if (confirm("Are you sure you want to delete+ " + element.firstName + " " + element.lastName + "'s account. This process is not reversible") == true) {
      this.userService.deleteUserDoc(element.uid);
    }
  }

  /**
   * Function to make the given FirebaseUser an admin
   * 
   * @param element Object containing data of the Firebase User selected
   */
  onMakeAdmin(element: FirebaseUser) {
    if (confirm("Do you want to make " + element.firstName + " " + element.lastName + " an admin user?") == true) {
      this.userService.updateSpecificUserData(element.uid, {"admin":true});
    }
  }
}
