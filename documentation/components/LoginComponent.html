import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { FirebaseAuthService } from 'src/app/services/firebase-auth.service';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent implements OnInit {

  public formLogin!: FormGroup;
  public formRegister!: FormGroup;
  public formForgotPassword!: FormGroup;

  public formError: boolean = false;
  public registerErrorDuplicate: boolean = false; //only when email is taken

  public formSubmit: boolean = false;

  public isRegistration: boolean = false;
  public forgotPassword: boolean = false;

  constructor(
    private router: Router,
    private fb: FormBuilder,
    private firebaseAuthService: FirebaseAuthService
  ) { }

  /**
   * Initializes three different forms with their validators to be used:
   * formLogin- The form to be used when a returning user is attempting to log in to the archive.
   * formRegister- The form to be used when a new user is attempting to register an account in our database.
   * formForgotPassword- The form to be used when a returning user has forgotten their password and is attempting to reset it.
   */
  ngOnInit(): void {
    this.formLogin = this.fb.group({
      loginEmail: ['', Validators.email],
      loginPassword: ['', Validators.required]
    });

    this.formRegister = this.fb.group({
      registerEmail: ['', Validators.email],
      registerPassword: ['', Validators.compose([Validators.required, Validators.minLength(6)])]
    });

    this.formForgotPassword = this.fb.group({
      knownEmail: ['', Validators.email]
    })
  }
  /**
   * On the submission of the login form, this methods gets the values from the form and checks each for validation.
   * If all inputs meet their condition(s), this method calls the login method.
   */
  //Check if the forms fields are valid. If so, try to login with the auth service.
  public onLoginSubmit() {

    this.formError = false;
    this.formSubmit = false;

    if (this.formLogin.valid) {
      const email = this.formLogin.get('loginEmail')?.value;
      const password = this.formLogin.get('loginPassword')?.value;

      this.login(email, password);
    } else {
      this.formSubmit = true;
    }
  }

  /**
   * This method calls the firebaseAuthSerivce "SignIn" method, using the email and password string passed from the formLogin form to allow the user access to the archive database of our website if they are an autheticated user. 
   * @param email The email of the user who is attempting to login to the archive portion of the website.
   * @param password  The password of the user who is attempting to login to the archive portion of the website.
   */
  public login(email: string, password: string) {
    this.firebaseAuthService.SignIn(email, password).catch((error) => {
      console.log(error);
      this.formError = true;
    });
  }
  /**
   * On submission of the formRegister form, this method checks each input for validation. If all inputs meet their condition(s), this method initializes email and password variables to be passed to the register method.  
   */
  public onRegistrationSubmit() {

    this.formError = false;
    this.registerErrorDuplicate = false;

    this.formSubmit = false;

    if (this.formRegister.valid) {
      const email = this.formRegister.get('registerEmail')?.value;
      const password = this.formRegister.get('registerPassword')?.value;

      this.register(email, password);
    } else {
      this.formSubmit = true;
    }
  }

  /**
   * This method calls the firebaseAuthService "SignUp" method, adding both the new user's email and password to the users collection of our database.
   * If the registering user's email is found in our database, this method throws an error and prevents signup with a duplicate email.
   * @param email The email of the new user who is registering.
   * @param password The password of the new user who is registering.
   */
  public register(email: string, password: string) {

    this.firebaseAuthService.SignUp(email, password).catch((error) => {

      if (error.code == "auth/email-already-in-use") {
        this.registerErrorDuplicate = true;
      } else {
        this.formError = true;
      }

    })
  }

  /**
   * On submission of the formForgotPassword form, this method checks if the input has met the necessary validations. 
   * If they have, the firebaseAuthService method "ForgotPassword" is called, sending an email to the user's input and outputting to the window a message to notify the user of the sent email.
   */
  public onForgotPasswordSubmit() {
    this.formError = false;
    this.formSubmit = false;

    if (this.formForgotPassword.valid) {
      const email = this.formForgotPassword.get('knownEmail')?.value;

      this.firebaseAuthService.ForgotPassword(email).then((result) => {
        window.alert('Password reset email sent, check your inbox for instructions.');
      }).catch((error) => {
        this.formError = true;
      });

    } else {
      this.formSubmit = true;
    }
  }
  /**
   * This method is called when: 
   *  1. Clicking the "Create An Account" button when logging in to go to the Registration page for new users. 
   *  2. Clicking the "Login Here!" button when registering to go to the Login page for existing users. 
   */
  public toggleRegistration() {
    this.formError = false;
    this.registerErrorDuplicate = false;

    this.isRegistration = !this.isRegistration;
    this.forgotPassword = false;

  }

  /**
   * The method is called when clicking on the "Forgot Password?" button on the login page to route the user to the ForgotPassword page to reset their password.  
   */
  public toggleForgotPassword() {
    this.forgotPassword = !this.forgotPassword;

    this.formError = false;
    this.registerErrorDuplicate = false;
  }

}
