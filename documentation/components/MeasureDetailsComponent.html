import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { Observable, Subscription } from 'rxjs';
import { Measure } from 'src/app/interfaces/measure';
import { MeasureFilesService } from 'src/app/services/measure-files.service';
import { MeasuresService } from 'src/app/services/measures.service';
import { AnalyticsService } from 'src/app/services/analytics.service';
import { FirebaseAuthService } from 'src/app/services/firebase-auth.service';

@Component({
  selector: 'app-measure-details',
  templateUrl: './measure-details.component.html',
  styleUrls: ['./measure-details.component.scss']
})
export class MeasureDetailsComponent implements OnInit {

  public measure!: Observable<Measure | undefined>;
  public measureId: string = "";
  public measureSub!: Subscription;
  public tags: string = "";

  constructor(private _activatedRoute: ActivatedRoute,
    private router: Router,
    private measuresService: MeasuresService,
    public measureFileService: MeasureFilesService, 
    private analytics: AnalyticsService,
    private authService: FirebaseAuthService) { }


  /**
   * This is test documentation for our ngOnInit life cycle hook.
   * 
   * Grabs the route parameters to get the measure link id, grabs the corresponding measure from the measures service, and redirects if the measure doesn't exist.
   */
  ngOnInit(): void {
    this._activatedRoute.paramMap.subscribe(params => {

      this.measureId = params.get('measureId')!;

      this.measure = this.measuresService.getMeasure(this.measureId);
      if (!this.measure) {
        this.router.navigate(['archive/measures']);
      }
    });

    this.measureSub = this.measure.subscribe((measure) => {
      if (measure) {
        for (let i = 0; i < measure.tags.length; i++) {
          this.tags += measure.tags[i];
          if (i != measure.tags.length - 1) this.tags += ", ";
        }

        //console.log(measure);

        this.analytics.track({
          'measureId': this.measureId,
          'userId': this.authService.id,
          'event' : 'measureViewed',
          'timestamp' : Date.now(),
        });
      }
      //console.log("TAGS: " + this.tags);
    });
  }

  ngOnDestroy() {
    this.measureSub.unsubscribe();
  }

  /**
   * Function called when links are clicked on to download files such as PDFs, DOCXs, etc.
   * 
   * @param file The filename string to ping the database and download the corresponding files.
   */
  onClick(file: string) {
    this.measureFileService.getLink(this.measureId + "/" + file).subscribe((url) => {
      window.open(url, "_blank");
    });
  }

}
